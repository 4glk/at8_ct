249 /******************************************************************************/
250 void IND_Update(void)
251 {
252  static uint8_t count = 0;
253  uint8_t i;
254 
255 #ifdef IND_SPI_CONNECTION
256  /*________________ подключение по SPI____________________________*/
257 
258  #ifdef IND_SHIFT_REG_DIG2
259  uint16_t tmp;
260  #else
261  uint8_t tmp;
262  #endif
263 
264  /*гасим индикатор*/
265  IND_LightOutSeg(spi_buf[IND_SHIFT_REG_SEG],IND_LIGHT_OUT);
266 
267  tmp = 0;
268  for (i = 0; i < IND_AMOUNT_NUM; i++){
269  tmp |= (1<<read_byte_flash(digit[i]));
270  }
271  IND_LightOutDig(spi_buf[IND_SHIFT_REG_DIG1],((uint8_t)tmp));
272 
273  #ifdef IND_SHIFT_REG_DIG2
274  IND_LightOutDig(spi_buf[IND_SHIFT_REG_DIG2],(tmp>>8));
275  #endif
276 
277  SPI_WriteArray(IND_AMOUNT_REG, spi_buf);
278 
279  #ifdef IND_SCAN_SEGMENT
280  /*___________реализация посегментной развертки___________*/
281 
282  tmp = 0;
283  for (i = 0; i < IND_AMOUNT_NUM; i++){
284  if (ind_buf[i]&read_byte_flash(mask[count])) {
285  tmp |= (1<<read_byte_flash(digit[i]));
286  }
287  }
288  IND_OutDig(spi_buf[IND_SHIFT_REG_DIG1], ((uint8_t)tmp));
289 
290  #ifdef IND_SHIFT_REG_DIG2
291  IND_OutDig(spi_buf[IND_SHIFT_REG_DIG2],(tmp>>8));
292  #endif
293 
294  IND_OutSeg(spi_buf[IND_SHIFT_REG_SEG], read_byte_flash(mask[count]));
295 
296  count++;
297  count &= 0x07;
298 
299  #else
300  /*___________реализация поразрядной развертки___________*/
301 
302  tmp = 1<<read_byte_flash(digit[count]);
303  IND_OutDig(spi_buf[IND_SHIFT_REG_DIG1], ((uint8_t)tmp));
304 
305  #ifdef IND_SHIFT_REG_DIG2
306  IND_OutDig(spi_buf[IND_SHIFT_REG_DIG2],(tmp>>8));
307  #endif
308 
309  IND_OutSeg(spi_buf[IND_SHIFT_REG_SEG], ind_buf[count]);
310 
311  count++;
312  if (count >= IND_AMOUNT_NUM){
313  count = 0;
314  }
315  #endif
316 
317  /*вывод на индикатор*/
318  SPI_WriteArray(IND_AMOUNT_REG, spi_buf);
319 
320 #else
321  /*____________________обычное подключение____________________________*/
322 
323  uint8_t tmp;
324 
325  /*гасим все индикаторы*/
326  IND_LightOutSeg(IND_PORT_SEG, IND_LIGHT_OUT);
327 
328  tmp = 0;
329  for (i = 0; i < IND_AMOUNT_NUM; i++){
330  tmp |= (1<<read_byte_flash(digit[i]));
331  }
332  IND_LightOutDig(IND_PORT_DIG, tmp);
333 
334  #ifdef IND_SCAN_SEGMENT
335  /*____________реализация посегментной развертки___________*/
336 
337  tmp = 0;
338 
339  for (i = 0; i < IND_AMOUNT_NUM; i++){
340  if (ind_buf[i]&read_byte_flash(mask[count])) {
341  tmp |= (1<<read_byte_flash(digit[i]));
342  }
343  }
344 
345  IND_OutDig(IND_PORT_DIG, tmp);
346  IND_OutSeg(IND_PORT_SEG, read_byte_flash(mask[count]));
347 
348  count++;
349  count &= 0x07;
350 
351  #else
352  /*____________реализаци поразрядной развертки____________*/
353 
354  IND_OutDig(IND_PORT_DIG, (1<<read_byte_flash(digit[count])));
355  IND_OutSeg(IND_PORT_SEG, ind_buf[count]);
356 
357  count++;
358  if (count >= IND_AMOUNT_NUM){
359  count = 0;
360  }
361  #endif
362 #endif
363  AddTask(IND_Update,5);
364 }
